## Structured Logging

Production scripts need timestamps, log levels, and output to both console and file. When something fails at 3 AM, the log file is all you have:

```bash
#!/bin/bash

LOG_FILE="/var/log/myapp/script.log"
LOG_LEVEL="${LOG_LEVEL:-info}"

# Log level hierarchy: debug < info < warn < error
declare -A LOG_LEVELS=([debug]=0 [info]=1 [warn]=2 [error]=3)

log() {
    local level="$1"; shift
    local message="$*"

    # Filter by configured level
    local level_num="${LOG_LEVELS[$level]:-1}"
    local threshold="${LOG_LEVELS[$LOG_LEVEL]:-1}"
    [ "$level_num" -lt "$threshold" ] && return 0

    local timestamp
    timestamp=$(date -u '+%Y-%m-%dT%H:%M:%SZ')
    local caller="${FUNCNAME[1]:-main}"
    local line="${BASH_LINENO[0]:-0}"

    # Structured format for log aggregators (Splunk, ELK, etc.)
    local entry="$timestamp level=$level caller=$caller line=$line msg=\"$message\""

    # Always write to file
    echo "$entry" >> "$LOG_FILE" 2>/dev/null

    # Write to console with color
    local color="" reset='\033[0m'
    case "$level" in
        debug) color='\033[36m' ;;   # cyan
        info)  color='\033[32m' ;;   # green
        warn)  color='\033[33m' ;;   # yellow
        error) color='\033[31m' ;;   # red
    esac
    printf "${color}[%-5s]${reset} %s\n" "${level^^}" "$message" >&2
}

# Convenience functions
debug() { log debug "$@"; }
info()  { log info "$@"; }
warn()  { log warn "$@"; }
error() { log error "$@"; }
fatal() { log error "$@"; exit 1; }
```

### Syslog Integration with logger

For scripts that run as system services, integrate with syslog so logs appear in the system journal:

```bash
#!/bin/bash

# Send to syslog (appears in journalctl and /var/log/syslog)
log_syslog() {
    local level="$1"; shift
    local message="$*"

    # logger sends to syslog daemon
    logger -t "myapp-script" -p "user.${level}" "$message"

    # Also print to console
    echo "[$(date '+%H:%M:%S')] [$level] $message"
}

log_syslog info "Backup started"
log_syslog err "Database connection failed"
# View with: journalctl -t myapp-script
```

### Log Rotation Within Scripts

For long-running scripts that manage their own logs:

```bash
#!/bin/bash

LOG_FILE="/var/log/myapp/script.log"
MAX_LOG_SIZE=$((10 * 1024 * 1024))  # 10 MB
MAX_LOG_FILES=5

rotate_log_if_needed() {
    [ ! -f "$LOG_FILE" ] && return 0

    local size
    size=$(stat -c%s "$LOG_FILE" 2>/dev/null || echo 0)
    [ "$size" -lt "$MAX_LOG_SIZE" ] && return 0

    # Shift existing rotated logs
    for ((i = MAX_LOG_FILES - 1; i > 0; i--)); do
        [ -f "${LOG_FILE}.$i" ] && mv "${LOG_FILE}.$i" "${LOG_FILE}.$((i + 1))"
    done

    mv "$LOG_FILE" "${LOG_FILE}.1"
    touch "$LOG_FILE"
}
```

<Callout type="tip" title="Key=Value Log Format">
Writing logs in `key=value` format (also called logfmt) makes them trivially parseable by log aggregation tools like Splunk, Elasticsearch, and Grafana Loki. Every field can be searched and filtered independently. Avoid free-form prose in production log messages.
</Callout>

## Configuration Management

Hard-coded values do not belong in scripts. Load them from config files with validation:

### Config Files with source

```bash
#!/bin/bash
set -euo pipefail

CONFIG_FILE="${CONFIG_FILE:-/etc/myapp/config.conf}"

# Defaults (overridden by config file)
APP_PORT=8080
APP_ENV="production"
DB_HOST="localhost"
DB_PORT=5432
LOG_LEVEL="info"
BACKUP_RETAIN_DAYS=30

load_config() {
    local config_file="$1"

    if [ ! -f "$config_file" ]; then
        echo "WARNING: Config not found: $config_file (using defaults)" >&2
        return 0
    fi

    # Security: reject config files with shell injection patterns
    if grep -qP '[\$`|;&]|\$\(' "$config_file"; then
        echo "ERROR: Config contains unsafe characters" >&2
        return 1
    fi

    # Parse only known KEY=VALUE entries
    while IFS='=' read -r key value; do
        [[ "$key" =~ ^[[:space:]]*# ]] && continue
        [[ -z "${key// /}" ]] && continue

        key="${key// /}"
        value="${value## }"; value="${value%% }"
        value="${value#\"}"; value="${value%\"}"

        case "$key" in
            APP_PORT|APP_ENV|DB_HOST|DB_PORT|LOG_LEVEL|BACKUP_RETAIN_DAYS)
                declare -g "$key=$value"
                ;;
            *)
                echo "WARNING: Unknown config key: $key" >&2
                ;;
        esac
    done < "$config_file"
}
```

### Environment Variables and .env Files

```bash
#!/bin/bash

# Precedence order (highest to lowest):
# 1. Command-line arguments
# 2. Environment variables
# 3. .env file
# 4. Config file
# 5. Script defaults

load_env_file() {
    local env_file="${1:-.env}"
    [ ! -f "$env_file" ] && return 0

    # Security check
    if grep -qP '[\$`|;&]' "$env_file"; then
        echo "ERROR: .env file contains unsafe characters" >&2
        return 1
    fi

    while IFS='=' read -r key value; do
        [[ "$key" =~ ^[[:space:]]*# ]] && continue
        [[ -z "${key// /}" ]] && continue

        key="${key// /}"
        value="${value## }"; value="${value%% }"
        value="${value#\"}"; value="${value%\"}"

        # Only set if not already set in environment
        # (environment variables take precedence over .env)
        if [ -z "${!key:-}" ]; then
            export "$key=$value"
        fi
    done < "$env_file"
}

# Usage
load_env_file ".env"
APP_PORT="${APP_PORT:-8080}"      # Env var > .env > default
DB_HOST="${DB_HOST:-localhost}"
```

### Config Validation

Always validate configuration before using it:

```bash
validate_config() {
    local errors=0

    # Port must be a number between 1 and 65535
    if ! [[ "$APP_PORT" =~ ^[0-9]+$ ]] || [ "$APP_PORT" -lt 1 ] || [ "$APP_PORT" -gt 65535 ]; then
        echo "ERROR: APP_PORT must be 1-65535 (got: $APP_PORT)" >&2
        errors=$((errors + 1))
    fi

    # Environment must be one of the known values
    case "$APP_ENV" in
        production|staging|development) ;;
        *) echo "ERROR: Invalid APP_ENV: $APP_ENV" >&2; errors=$((errors + 1)) ;;
    esac

    # Log level must be valid
    case "$LOG_LEVEL" in
        debug|info|warn|error) ;;
        *) echo "ERROR: Invalid LOG_LEVEL: $LOG_LEVEL" >&2; errors=$((errors + 1)) ;;
    esac

    # Retention must be positive
    if ! [[ "$BACKUP_RETAIN_DAYS" =~ ^[0-9]+$ ]] || [ "$BACKUP_RETAIN_DAYS" -lt 1 ]; then
        echo "ERROR: BACKUP_RETAIN_DAYS must be positive (got: $BACKUP_RETAIN_DAYS)" >&2
        errors=$((errors + 1))
    fi

    return "$errors"
}
```

<Callout type="warning" title="Never source Untrusted Config Files">
The `source` command executes a file as bash code. If someone puts `$(rm -rf /)` in your config file, `source` will run it. Always parse config files line-by-line with `read`, validating each key and value. Only accept known keys. Reject any file containing shell metacharacters like `$`, backticks, `|`, or `;`.
</Callout>

### The Main Function Pattern

Always wrap your script logic in a `main()` function. This keeps the global scope clean and makes the script sourceable for testing:

```bash
#!/bin/bash
set -euo pipefail

# Source libraries
SCRIPT_DIR=$(cd "$(dirname "$0")" && pwd)
source "$SCRIPT_DIR/../lib/logging.sh"
source "$SCRIPT_DIR/../lib/config.sh"

# Default configuration
APP_NAME="myservice"
VERSION="1.0.0"

do_work() {
    info "Doing the actual work..."
    # ...
}

main() {
    info "Starting $APP_NAME v$VERSION"
    load_config "${CONFIG_FILE:-}"
    validate_config
    do_work
    info "Complete"
}

# Only run main if this script is executed, not sourced
# This allows tests to source the file and call individual functions
if [[ "${BASH_SOURCE[0]}" == "${0}" ]]; then
    main "$@"
fi
```

### Sourcing Libraries

```bash
#!/bin/bash

# lib/logging.sh — reusable across all scripts
LOG_LEVEL="${LOG_LEVEL:-info}"

log_info()  { echo "[$(date '+%H:%M:%S')] [INFO]  $*"; }
log_warn()  { echo "[$(date '+%H:%M:%S')] [WARN]  $*" >&2; }
log_error() { echo "[$(date '+%H:%M:%S')] [ERROR] $*" >&2; }

require_command() {
    if ! command -v "$1" &>/dev/null; then
        log_error "Required command not found: $1"
        return 1
    fi
}

require_file() {
    if [ ! -f "$1" ]; then
        log_error "Required file not found: $1"
        return 1
    fi
}
```

## Testing Scripts

### Unit Testing Functions

The simplest testing approach: write assertion functions and test each function in isolation:

```bash
#!/bin/bash

# test_utils.sh — simple test framework

TESTS_RUN=0
TESTS_PASSED=0
TESTS_FAILED=0

assert_equals() {
    local expected="$1"
    local actual="$2"
    local message="${3:-assertion}"
    TESTS_RUN=$((TESTS_RUN + 1))

    if [ "$expected" = "$actual" ]; then
        echo "  [PASS] $message"
        TESTS_PASSED=$((TESTS_PASSED + 1))
    else
        echo "  [FAIL] $message"
        echo "         Expected: '$expected'"
        echo "         Actual:   '$actual'"
        TESTS_FAILED=$((TESTS_FAILED + 1))
    fi
}

assert_exit_code() {
    local expected_code="$1"
    shift
    local actual_code

    "$@" > /dev/null 2>&1
    actual_code=$?
    TESTS_RUN=$((TESTS_RUN + 1))

    if [ "$expected_code" -eq "$actual_code" ]; then
        echo "  [PASS] exit code $expected_code: $*"
        TESTS_PASSED=$((TESTS_PASSED + 1))
    else
        echo "  [FAIL] exit code: expected $expected_code, got $actual_code: $*"
        TESTS_FAILED=$((TESTS_FAILED + 1))
    fi
}

assert_contains() {
    local haystack="$1"
    local needle="$2"
    local message="${3:-contains check}"
    TESTS_RUN=$((TESTS_RUN + 1))

    if echo "$haystack" | grep -qF "$needle"; then
        echo "  [PASS] $message"
        TESTS_PASSED=$((TESTS_PASSED + 1))
    else
        echo "  [FAIL] $message"
        echo "         String does not contain: '$needle'"
        TESTS_FAILED=$((TESTS_FAILED + 1))
    fi
}

test_summary() {
    echo ""
    echo "Results: $TESTS_PASSED/$TESTS_RUN passed, $TESTS_FAILED failed"
    return "$TESTS_FAILED"
}
```

### The bats Testing Framework

[bats](https://github.com/bats-core/bats-core) (Bash Automated Testing System) is the standard framework for bash testing:

```bash
#!/usr/bin/env bats
# test_deploy.bats — run with: bats test_deploy.bats

# Load the script's functions (without running main)
setup() {
    source ./deploy.sh  # Works because of the BASH_SOURCE guard
    DEPLOY_DIR=$(mktemp -d)
}

teardown() {
    rm -rf "$DEPLOY_DIR"
}

@test "create_release creates the release directory" {
    create_release "v1.0.0"
    [ -d "$DEPLOY_DIR/releases/v1.0.0" ]
}

@test "activate_release switches the symlink" {
    mkdir -p "$DEPLOY_DIR/releases/v1.0.0"
    activate_release "$DEPLOY_DIR/releases/v1.0.0"
    [ -L "$DEPLOY_DIR/current" ]
    [ "$(readlink -f "$DEPLOY_DIR/current")" = "$DEPLOY_DIR/releases/v1.0.0" ]
}

@test "health_check fails on invalid URL" {
    run health_check "http://localhost:99999/nonexistent" 2
    [ "$status" -ne 0 ]
}

@test "rollback restores previous release" {
    mkdir -p "$DEPLOY_DIR/releases/v1.0.0" "$DEPLOY_DIR/releases/v2.0.0"
    activate_release "$DEPLOY_DIR/releases/v1.0.0"
    activate_release "$DEPLOY_DIR/releases/v2.0.0"
    rollback
    [ "$(readlink -f "$DEPLOY_DIR/current")" = "$DEPLOY_DIR/releases/v1.0.0" ]
}
```

<Callout type="tip" title="Test Helpers">
Create test helper functions for common operations: setting up temp directories, creating mock config files, capturing output. Good test helpers make tests readable and reduce boilerplate. The `setup()` and `teardown()` functions in bats run before and after each test automatically.
</Callout>

## Documentation

### Usage and Help Functions

Every script needs a `-h` or `--help` flag:

```bash
#!/bin/bash

SCRIPT_NAME=$(basename "$0")
VERSION="1.2.0"

usage() {
    cat << EOF
$SCRIPT_NAME v$VERSION — Automated deployment tool

USAGE:
    $SCRIPT_NAME [OPTIONS] COMMAND [ARGS]

COMMANDS:
    deploy VERSION      Deploy a specific version
    rollback            Roll back to the previous version
    status              Show current deployment status

    -e, --env ENV       Target environment (default: staging)
    -d, --dry-run       Show what would happen without doing it
    -f, --force         Skip confirmation prompts
    -v, --verbose       Enable debug logging
    -c, --config FILE   Use a specific config file
    -h, --help          Show this help message
    -V, --version       Show version

ENVIRONMENT VARIABLES:
    DEPLOY_ENV          Same as --env
    CONFIG_FILE         Same as --config
    LOG_LEVEL           Set log level (debug, info, warn, error)

EXAMPLES:

    0   Success
```

### Inline Comments: When and How

```bash
# BAD: Comments that restate the code
count=$((count + 1))  # Increment count by 1

# GOOD: Comments that explain WHY
# Retry with exponential backoff — the API rate-limits after 10 requests/second
sleep $((2 ** attempt))

# GOOD: Comments that explain a non-obvious decision
# We use process substitution here instead of a pipe to avoid
# the subshell variable trap (COUNT would be 0 after the loop)
while read -r line; do
    COUNT=$((COUNT + 1))
done < <(find /etc -type f)

# GOOD: Comments that warn about gotchas
# WARNING: This file is sourced by deploy.sh — do not add executable code here
# WARNING: Changing DB_HOST requires restarting all workers
```

## Security Best Practices

### Never Store Secrets in Scripts

```bash
# TERRIBLE: Password in the script
# DB_PASSWORD="super_secret_123"

# BAD: Password in a config file checked into git
# source config.sh  # Contains DB_PASSWORD=...

# GOOD: Read from environment (set by secrets manager, vault, etc.)
DB_PASSWORD="${DB_PASSWORD:?ERROR: DB_PASSWORD not set}"

# GOOD: Read from a restricted file
DB_PASSWORD=$(cat /run/secrets/db_password)

# GOOD: Use a secrets manager CLI
# DB_PASSWORD=$(vault kv get -field=password secret/myapp/db)
```

### Validate All Input

```bash
#!/bin/bash
set -euo pipefail

# Validate user-supplied values BEFORE using them
validate_input() {
    local input="$1"
    local name="$2"

    # Reject empty input
    if [ -z "$input" ]; then
        echo "ERROR: $name cannot be empty" >&2
        return 1
    fi

    # Reject input with shell metacharacters
    if [[ "$input" =~ [^a-zA-Z0-9._-] ]]; then
        echo "ERROR: $name contains invalid characters: $input" >&2
        return 1
    fi

    # Reject path traversal attempts
    if [[ "$input" == *".."* ]]; then
        echo "ERROR: $name contains path traversal: $input" >&2
        return 1
    fi
}

# Usage
validate_input "${1:-}" "version" || exit 2
validate_input "${2:-}" "environment" || exit 2
```

### Avoid eval

```bash
# DANGEROUS: eval executes arbitrary code
# user_input="; rm -rf /"
# eval "echo $user_input"          # Runs rm -rf /!

# SAFE: Use arrays to build dynamic commands
cmd=("rsync" "-a" "--delete")
if [ "${VERBOSE:-false}" = "true" ]; then
    cmd+=("--verbose")
fi
cmd+=("$SOURCE" "$DEST")
"${cmd[@]}"  # Each array element is a separate argument — safe

# SAFE: Use indirect references instead of eval
var_name="APP_PORT"
echo "${!var_name}"  # Reads APP_PORT's value — no eval needed
```

### Quote Everything

```bash
# BROKEN: Unquoted variables break on spaces and glob characters
filename="my file.txt"
# rm $filename          # Tries to remove "my" and "file.txt" separately!
# rm $filename          # Worse: if filename is "*.txt", deletes ALL .txt files!

# SAFE: Always quote variables
rm "$filename"          # Correctly removes "my file.txt"

# SAFE: Quote arrays with [@]
files=("file 1.txt" "file 2.txt" "*.txt")
for f in "${files[@]}"; do     # Each element is one item, even with spaces
    echo "Processing: $f"
done
```

### Use mktemp for Temp Files

```bash
# DANGEROUS: Predictable path — symlink attack vector
# temp="/tmp/myapp.tmp"

# SAFE: Unpredictable, unique path
TEMP_FILE=$(mktemp /tmp/myapp.XXXXXXXXXX)
TEMP_DIR=$(mktemp -d /tmp/myapp.XXXXXXXXXX)
trap 'rm -rf "$TEMP_FILE" "$TEMP_DIR"' EXIT

# EVEN SAFER: Use process substitution when possible (no temp files at all)
diff <(sort file1.txt) <(sort file2.txt)
```

### Set Restrictive Permissions

```bash
#!/bin/bash

# Set umask at the top of every production script
umask 077  # New files: owner read/write only

# Validate file ownership before sourcing
validate_file() {
    local file="$1"
    local owner
    owner=$(stat -c '%U' "$file")

    if [ "$owner" != "root" ] && [ "$owner" != "$(whoami)" ]; then
        echo "ERROR: $file owned by $owner (expected root or $(whoami))" >&2
        return 1
    fi

    local perms
    perms=$(stat -c '%a' "$file")
    if [[ "$perms" =~ [2367][2367]$ ]]; then
        echo "ERROR: $file is group/world writable (perms: $perms)" >&2
        return 1
    fi
}
```

## Performance

### Avoid Unnecessary Subshells

```bash
# SLOW: Subshell for simple variable assignment
result=$(echo "$input" | tr '[:lower:]' '[:upper:]')

# FAST: Use bash built-in string manipulation
result="${input^^}"

# SLOW: External command for basename
dir=$(dirname "$filepath")
base=$(basename "$filepath")

# FAST: Parameter expansion
dir="${filepath%/*}"
base="${filepath##*/}"

# SLOW: cat into a pipeline
cat file.txt | grep "pattern"

# FAST: Redirect directly
grep "pattern" file.txt
```

### Use Built-ins Over External Commands

```bash
# SLOW: External command for substring check
if echo "$string" | grep -q "pattern"; then

# FAST: Bash built-in pattern matching
if [[ "$string" == *"pattern"* ]]; then

# SLOW: External command for arithmetic
result=$(expr $a + $b)

# FAST: Bash arithmetic
result=$((a + b))

# SLOW: External command to count array elements
count=$(echo "${array[@]}" | wc -w)

# FAST: Bash built-in
count=${#array[@]}
```

### Parallel Execution

```bash
#!/bin/bash

# Process files in parallel with a worker pool
MAX_PARALLEL=4
RUNNING=0

process_file() {
    local file="$1"
    # ... do work ...
    sleep 0.5
}

for file in /etc/*.conf; do
    process_file "$file" &
    RUNNING=$((RUNNING + 1))

    if [ "$RUNNING" -ge "$MAX_PARALLEL" ]; then
        wait -n 2>/dev/null || true
        RUNNING=$((RUNNING - 1))
    fi
done

wait  # Wait for remaining workers
```

## Production-Ready Checklist

Before deploying any script to production, verify:

```bash
# The checklist every script should pass:

# 1. Strict mode
set -euo pipefail

# 2. Proper logging with timestamps and levels
# (not just echo statements)

# 3. Configuration externalized (not hard-coded)
# CONFIG_FILE, environment variables, defaults

# 4. Cleanup on exit
trap cleanup EXIT

# 5. Lock file for scripts that shouldn't run concurrently
flock -n 200 || exit 0

# 6. Input validation — every user-supplied value checked

# 7. Usage/help text with -h flag

# 8. Meaningful exit codes (0=success, 1=error, 2=bad args)

# 9. Dry-run mode for destructive operations

# 10. All variables quoted ("$var" not $var)

# 11. mktemp for temporary files, never hardcoded paths

# 12. No secrets in the script or config files in version control

# 13. Tested — at least manual tests, ideally automated

# 14. Idempotent — safe to run twice without side effects
```

<Exercise number={1} title="Production-Grade Script">

```bash
cat > ~/prod-grade.sh << 'SCRIPT'
#!/bin/bash
set -euo pipefail

# Production-Grade Script Exercise
# Take a basic script and add logging, config, error handling, and help

readonly SCRIPT_NAME=$(basename "$0")
readonly SCRIPT_VERSION="1.0.0"

# --- Defaults ---
CONFIG_FILE=""
LOG_FILE="/tmp/prod-grade.log"
LOG_LEVEL="info"
DRY_RUN=false
VERBOSE=false
LOCK_FILE="/tmp/prod-grade.lock"
OUTPUT_DIR="/tmp/prod-grade-output"

# --- Logging ---

declare -A LOG_LEVELS=([debug]=0 [info]=1 [warn]=2 [error]=3)

_log() {
    local level="$1"; shift
    local ln="${LOG_LEVELS[$level]:-1}"
    local thr="${LOG_LEVELS[$LOG_LEVEL]:-1}"
    [ "$ln" -lt "$thr" ] && return 0

    local ts
    ts=$(date '+%Y-%m-%d %H:%M:%S')
    local msg="[$ts] [${level^^}] $*"
    echo "$msg" >> "$LOG_FILE" 2>/dev/null

    case "$level" in
        error|warn) echo "$msg" >&2 ;;
        *)          echo "$msg" ;;
    esac
}

debug() { _log debug "$@"; }
info()  { _log info "$@"; }
warn()  { _log warn "$@"; }
error() { _log error "$@"; }
fatal() { _log error "$@"; exit 1; }

# --- Lock file ---

acquire_lock() {
    exec 200>"$LOCK_FILE"
    if ! flock -n 200; then
        fatal "Another instance is running (lock: $LOCK_FILE)"
    fi
    echo $$ > "$LOCK_FILE"
    debug "Lock acquired (PID: $$)"
}

# --- Config loading ---

load_config() {
    local f="$1"
    [ ! -f "$f" ] && { error "Config not found: $f"; return 1; }

    if grep -qP '[\$`|;&]' "$f"; then
        error "Config contains unsafe characters"
        return 1
    fi

    info "Loading config: $f"
    while IFS='=' read -r key value; do
        [[ "$key" =~ ^[[:space:]]*# ]] && continue
        [[ -z "${key// /}" ]] && continue
        key="${key// /}"; value="${value## }"; value="${value%% }"
        value="${value#\"}"; value="${value%\"}"

        case "$key" in
            LOG_LEVEL)   LOG_LEVEL="$value" ;;
            OUTPUT_DIR)  OUTPUT_DIR="$value" ;;
            *)           debug "Unknown config key: $key" ;;
        esac
    done < "$f"
}

# --- Cleanup ---

cleanup() {
    flock -u 200 2>/dev/null
    rm -f "$LOCK_FILE"
    debug "Cleanup complete"
}

# --- Usage ---

usage() {
    cat << EOF
$SCRIPT_NAME v$SCRIPT_VERSION — System report generator

USAGE:
    $SCRIPT_NAME [OPTIONS] COMMAND

COMMANDS:
    check       Run system checks
    report      Generate system report
    clean       Clean old reports

OPTIONS:
    -c FILE     Configuration file
    -l LEVEL    Log level: debug, info, warn, error (default: info)
    -o DIR      Output directory (default: $OUTPUT_DIR)
    -d          Dry run — show what would happen
    -v          Verbose (sets log level to debug)
    -h          Show this help
    -V          Show version

EXAMPLES:
    $SCRIPT_NAME check
    $SCRIPT_NAME -c /etc/myapp.conf report
    $SCRIPT_NAME -v -d clean
EOF
    exit "${1:-0}"
}

# --- Commands ---

cmd_check() {
    info "Running system checks"
    local passed=0 failed=0

    # Disk check
    local disk
    disk=$(df / | awk 'NR==2 {print $5}' | tr -d '%')
    if [ "$disk" -lt 90 ]; then
        info "  [PASS] Disk: ${disk}% used"; passed=$((passed + 1))
    else
        warn "  [FAIL] Disk: ${disk}% used"; failed=$((failed + 1))
    fi

    # Memory check
    local mem_avail
    mem_avail=$(awk '/MemAvailable/ {print int($2/1024)}' /proc/meminfo 2>/dev/null || echo 0)
    if [ "$mem_avail" -gt 100 ]; then
        info "  [PASS] Memory: ${mem_avail}MB available"; passed=$((passed + 1))
    else
        warn "  [FAIL] Memory: ${mem_avail}MB available"; failed=$((failed + 1))
    fi

    # Writable /tmp
    if [ -w /tmp ]; then
        info "  [PASS] /tmp is writable"; passed=$((passed + 1))
    else
        warn "  [FAIL] /tmp not writable"; failed=$((failed + 1))
    fi

    # Load average
    local load
    load=$(awk '{printf "%.0f", $1}' /proc/loadavg 2>/dev/null || echo 0)
    local cores
    cores=$(nproc 2>/dev/null || echo 1)
    if [ "$load" -le "$cores" ]; then
        info "  [PASS] Load: $load (cores: $cores)"; passed=$((passed + 1))
    else
        warn "  [WARN] Load: $load (cores: $cores)"; failed=$((failed + 1))
    fi

    info "Checks: $passed passed, $failed failed"
    return "$failed"
}

cmd_report() {
    info "Generating system report"

    if $DRY_RUN; then
        info "[DRY RUN] Would generate report in $OUTPUT_DIR"
        return 0
    fi

    mkdir -p "$OUTPUT_DIR"
    local report="$OUTPUT_DIR/report-$(date '+%Y%m%d-%H%M%S').txt"

    {
        echo "System Report"
        echo "Generated: $(date -u '+%Y-%m-%dT%H:%M:%SZ')"
        echo "Host:      $(hostname)"
        echo "Kernel:    $(uname -r)"
        echo ""
        echo "--- Resources ---"
        printf "  %-12s %s\n" "CPU Cores:" "$(nproc 2>/dev/null || echo N/A)"
        printf "  %-12s %s\n" "Memory:" "$(free -h 2>/dev/null | awk '/Mem:/ {print $3 "/" $2}' || echo N/A)"
        printf "  %-12s %s\n" "Disk (/):" "$(df -h / | awk 'NR==2 {print $3 "/" $2 " (" $5 " used)"}')"
        echo ""
        echo "--- Top Processes by CPU ---"
        ps aux --sort=-%cpu 2>/dev/null | head -6 || true
    } > "$report"

    info "Report saved: $report"
    cat "$report"
}

cmd_clean() {
    info "Cleaning old reports"

    if [ ! -d "$OUTPUT_DIR" ]; then
        info "Nothing to clean"
        return 0
    fi

    local old_files
    old_files=$(find "$OUTPUT_DIR" -name "report-*.txt" -mtime +7 2>/dev/null || true)

    if [ -z "$old_files" ]; then
        info "No old reports to clean"
        return 0
    fi

    local count
    count=$(echo "$old_files" | wc -l)

    if $DRY_RUN; then
        info "[DRY RUN] Would remove $count old reports"
    else
        echo "$old_files" | while read -r f; do
            rm -f "$f"
            debug "Removed: $f"
        done
        info "Removed $count old reports"
    fi
}

# --- Main ---

main() {
    while getopts ":c:l:o:dvhV" opt; do
        case "$opt" in
            c) CONFIG_FILE="$OPTARG" ;;
            l) LOG_LEVEL="$OPTARG" ;;
            o) OUTPUT_DIR="$OPTARG" ;;
            d) DRY_RUN=true ;;
            v) VERBOSE=true; LOG_LEVEL="debug" ;;
            h) usage 0 ;;
            V) echo "$SCRIPT_NAME v$SCRIPT_VERSION"; exit 0 ;;
            :) fatal "-$OPTARG requires an argument" ;;
            *) fatal "Unknown option: -$OPTARG" ;;
        esac
    done
    shift $((OPTIND - 1))

    [ -n "$CONFIG_FILE" ] && { load_config "$CONFIG_FILE" || exit 1; }

    local command="${1:-}"
    [ -z "$command" ] && { error "No command specified"; usage 1; }

    acquire_lock
    trap cleanup EXIT

    info "Starting $SCRIPT_NAME v$SCRIPT_VERSION (command: $command)"
    $DRY_RUN && info "DRY RUN MODE"

    case "$command" in
        check)  cmd_check ;;
        report) cmd_report ;;
        clean)  cmd_clean ;;
        *)      error "Unknown command: $command"; usage 1 ;;
    esac

    info "Done"
}

main "$@"
```

```bash
_log() {
    local level="$1"; shift
    local ln="${_LOG_LEVELS[$level]:-1}"
    local thr="${_LOG_LEVELS[$LOG_LEVEL]:-1}"
    [ "$ln" -lt "$thr" ] && return 0

    local ts
    ts=$(date -u '+%Y-%m-%dT%H:%M:%SZ')

    echo "$entry" >> "$LOG_FILE" 2>/dev/null

    local color="" reset='\033[0m'
    case "$level" in
        debug) color='\033[36m' ;; info) color='\033[32m' ;;
        warn)  color='\033[33m' ;; error) color='\033[31m' ;;
    esac
    printf "${color}[%-5s]${reset} %s\n" "${level^^}" "$*" >&2
}

debug() { _log debug "$@"; }
info()  { _log info "$@"; }
warn()  { _log warn "$@"; }
error() { _log error "$@"; }

acquire_lock() {
    mkdir -p "$(dirname "$LOCK_FILE")"
    exec 200>"$LOCK_FILE"
    if ! flock -n 200; then
        local other
        other=$(cat "$LOCK_FILE" 2>/dev/null || echo "unknown")
        error "Another instance is running (PID: $other)"
        exit $EXIT_LOCKED
    fi
    echo $$ > "$LOCK_FILE"

load_config() {
    local f="$1"
    [ ! -f "$f" ] && { error "Config not found: $f"; return 1; }

        error "Config contains unsafe characters: $f"
        return 1
    fi

    debug "Loading config: $f"
    while IFS='=' read -r key value; do
        [[ "$key" =~ ^[[:space:]]*# ]] && continue
        [[ -z "${key// /}" ]] && continue
        key="${key// /}"; value="${value## }"; value="${value%% }"
        value="${value#\"}"; value="${value%\"}"

        case "$key" in
            APP_NAME)     APP_NAME="$value" ;;
            ENVIRONMENT)  ENVIRONMENT="$value" ;;
            LOG_LEVEL)    LOG_LEVEL="$value" ;;
            WORK_DIR)     WORK_DIR="$value" ;;
            *)            debug "Unknown config key: $key" ;;
        esac
    done < "$f"

    info "Config loaded: $f"
}

cmd_deploy() {
    local version="${1:-}"
    if [ -z "$version" ]; then
        error "deploy requires a version argument"
        error "Usage: $PROG deploy VERSION"
        exit $EXIT_BAD_ARGS
    fi

    # Validate version format
    if ! [[ "$version" =~ ^v?[0-9]+\.[0-9]+\.[0-9]+$ ]]; then
        error "Invalid version format: $version (expected: v1.2.3 or 1.2.3)"
        exit $EXIT_BAD_ARGS
    fi

    local release_id
    release_id="$(date '+%Y%m%d%H%M%S')-${version}"
    local release_dir="$WORK_DIR/releases/$release_id"

    info "========================================="
    info "Deploying $APP_NAME $version"
    info "  Environment: $ENVIRONMENT"
    info "  Release ID:  $release_id"
    info "========================================="

    if $DRY_RUN; then
        info "[DRY RUN] Would create release: $release_dir"
        info "[DRY RUN] Would switch symlink: $WORK_DIR/current -> $release_dir"
        info "[DRY RUN] Would run health check"
        return 0
    fi

    # Ensure workspace exists
    if [ ! -d "$WORK_DIR/releases" ]; then
        warn "Workspace not initialized — running init first"
        cmd_init
    fi

    info "Creating release..."
    mkdir -p "$release_dir"/{bin,config,logs}
    echo "#!/bin/bash" > "$release_dir/bin/app"
    echo "echo '$APP_NAME $version running on \${APP_PORT:-8080}'" >> "$release_dir/bin/app"
    chmod +x "$release_dir/bin/app"

    cat > "$release_dir/VERSION" << EOF
version=$version
release_id=$release_id
environment=$ENVIRONMENT
deployed_at=$(date -u '+%Y-%m-%dT%H:%M:%SZ')
deployed_by=$(whoami)
EOF

    ln -sf "$WORK_DIR/shared/config/app.conf" "$release_dir/config/app.conf" 2>/dev/null || true

    # Save previous release for rollback
    if [ -L "$WORK_DIR/current" ]; then
        readlink -f "$WORK_DIR/current" > "$WORK_DIR/.previous_release"
        debug "Previous release saved"
    fi

    # Atomic symlink switch
    local tmp_link="$WORK_DIR/current.tmp.$$"
    ln -sfn "$release_dir" "$tmp_link"
    mv -Tf "$tmp_link" "$WORK_DIR/current"
    info "Activated: $release_dir"

    # Verify
    if [ -x "$WORK_DIR/current/bin/app" ]; then
        local output
        output=$("$WORK_DIR/current/bin/app")
        info "Health check: $output"
    fi

    local count
    count=$(ls -1 "$WORK_DIR/releases" 2>/dev/null | wc -l)
    if [ "$count" -gt 5 ]; then
        ls -1t "$WORK_DIR/releases" | tail -n +6 | while read -r old; do
            info "Removing old release: $old"
            rm -rf "${WORK_DIR:?}/releases/$old"
        done
    fi

    info "Deployment SUCCESSFUL: $APP_NAME $version"

cmd_status() {

    echo ""

    if [ -d "$WORK_DIR" ]; then
        echo "Workspace: EXISTS"

        if [ -L "$WORK_DIR/current" ]; then
            local current
            current=$(readlink -f "$WORK_DIR/current")
            echo "Current:   $current"

            if [ -f "$WORK_DIR/current/VERSION" ]; then
                echo ""
                echo "--- Version Info ---"
                cat "$WORK_DIR/current/VERSION" | sed 's/^/  /'
            fi
        else
            echo "Current:   (not deployed)"
        fi

        echo ""
        echo "--- Releases ---"
        if ls "$WORK_DIR/releases"/* &>/dev/null 2>&1; then
            ls -1t "$WORK_DIR/releases" | while read -r rel; do
                local marker=" "
                local rel_path="$WORK_DIR/releases/$rel"
                if [ -L "$WORK_DIR/current" ]; then
                    [ "$(readlink -f "$WORK_DIR/current")" = "$(readlink -f "$rel_path")" ] && marker="*"
                fi
                local size
                size=$(du -sh "$rel_path" 2>/dev/null | cut -f1 || echo "?")
                echo "  $marker $rel ($size)"
            done
        else
            echo "  (none)"
        fi
    else
        echo "Workspace: NOT INITIALIZED"
        echo "  Run: $PROG init"
    fi
}

cmd_logs() {
    local count="${1:-20}"

    if [ ! -f "$LOG_FILE" ]; then
        info "No log file found: $LOG_FILE"
        return 0
    fi

    echo "=== Last $count log entries ==="
    echo "Log file: $LOG_FILE"
    echo ""
    tail -n "$count" "$LOG_FILE"
}

    echo "  ENVIRONMENT:  $ENVIRONMENT"

    fi
}

main() {
    # Parse global options
    while getopts ":e:c:l:n:dvhV" opt; do
        case "$opt" in
            e) ENVIRONMENT="$OPTARG" ;;
            c) CONFIG_FILE="$OPTARG" ;;
            l) LOG_LEVEL="$OPTARG" ;;
            n) APP_NAME="$OPTARG" ;;
            d) DRY_RUN=true ;;
            v) LOG_LEVEL="debug" ;;
            h) usage $EXIT_OK ;;
            V) echo "$PROG v$VERSION"; exit $EXIT_OK ;;
            :) error "-$OPTARG requires an argument"; exit $EXIT_BAD_ARGS ;;
            *) error "Unknown option: -$OPTARG"; usage $EXIT_BAD_ARGS ;;
        esac
    done
    shift $((OPTIND - 1))

    # Load config if specified
    if [ -n "$CONFIG_FILE" ]; then
        load_config "$CONFIG_FILE" || exit $EXIT_ERROR
    fi
    validate_config || exit $EXIT_BAD_ARGS

    # Get subcommand
    local command="${1:-}"
    [ -z "$command" ] && { error "No command specified"; usage $EXIT_BAD_ARGS; }
    shift

    # Setup
    trap cleanup EXIT
    trap handle_signal SIGINT SIGTERM
    acquire_lock

    debug "Command: $command, Args: $*"

        status)  cmd_status ;;
        logs)    cmd_logs "$@" ;;
        config)  cmd_config ;;
        *)       error "Unknown command: $command"; usage $EXIT_BAD_ARGS ;;
    esac

    debug "$PROG completed successfully"
}
```

## Key Takeaways

- Structured logging with timestamps and levels makes debugging production incidents possible
- Write logs in `key=value` format for easy parsing by log aggregation tools
- Load configuration from files and environment variables, never hardcode values
- Validate every config key — reject unknown keys and unsafe characters
- The main function pattern (`if [[ "${BASH_SOURCE[0]}" == "${0}" ]]`) makes scripts testable
- Use `flock` for lock files — PID-file-only approaches have race conditions
- Write idempotent scripts that check state before making changes
- Never use `eval` with untrusted input — use arrays for dynamic commands
- Always quote variables and use `mktemp` for temporary files
- Set `umask 077` in security-sensitive scripts
- Test your functions — even a simple assert framework catches regressions
- Every script needs `-h` for help and meaningful exit codes
