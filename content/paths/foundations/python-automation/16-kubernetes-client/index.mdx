## Setting Up the Kubernetes Python Client

The official Kubernetes Python client is a full-featured SDK that mirrors the Kubernetes API. It can talk to any cluster your `kubectl` can reach.

```python
# Install the client
# pip install kubernetes

from kubernetes import client, config

# Load kubeconfig from default location (~/.kube/config)
# This is what you use in development / on your laptop
config.load_kube_config()

# In a pod running inside Kubernetes, use in-cluster config instead:
# config.load_incluster_config()

# Auto-detect: try in-cluster first, fall back to kubeconfig
def load_k8s_config():
    """Load Kubernetes config, auto-detecting the environment."""
    try:
        config.load_incluster_config()
        print("Loaded in-cluster config")
    except config.ConfigException:
        config.load_kube_config()
        print("Loaded kubeconfig")

load_k8s_config()

# Create API clients for different resource groups
v1 = client.CoreV1Api()          # Pods, Services, ConfigMaps, Secrets, Namespaces
apps_v1 = client.AppsV1Api()     # Deployments, StatefulSets, DaemonSets
batch_v1 = client.BatchV1Api()   # Jobs, CronJobs

# Verify connection
version = client.VersionApi().get_code()
print(f"Connected to Kubernetes {version.git_version}")
```

<Callout type="tip" title="Context Switching">
The `load_kube_config()` function accepts a `context` parameter to target a specific cluster: `config.load_kube_config(context="production-cluster")`. This is how your automation switches between dev, staging, and production clusters without changing environment variables.
</Callout>

## Working with Pods

Pods are the fundamental unit in Kubernetes. The Python client lets you list, inspect, read logs from, and exec into pods programmatically.

### Listing and Inspecting Pods

```python
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

# List all pods in a namespace
pods = v1.list_namespaced_pod(namespace="default")
print(f"Found {len(pods.items)} pods in 'default' namespace\n")

for pod in pods.items:
    name = pod.metadata.name
    phase = pod.status.phase  # Pending, Running, Succeeded, Failed, Unknown
    node = pod.spec.node_name or "unscheduled"

    # Get container statuses
    container_statuses = pod.status.container_statuses or []
    ready_count = sum(1 for cs in container_statuses if cs.ready)
    total_count = len(container_statuses)

    # Get restart count
    restart_count = sum(cs.restart_count for cs in container_statuses)

    print(f"  {name:45s}  {phase:10s}  {ready_count}/{total_count} ready  "
          f"restarts: {restart_count}  node: {node}")

# List pods across ALL namespaces
all_pods = v1.list_pod_for_all_namespaces()
print(f"\nTotal pods across all namespaces: {len(all_pods.items)}")

# Get a specific pod
pod = v1.read_namespaced_pod(name="my-pod", namespace="default")
print(f"\nPod: {pod.metadata.name}")
print(f"  Namespace: {pod.metadata.namespace}")
print(f"  Labels: {pod.metadata.labels}")
print(f"  IP: {pod.status.pod_ip}")
print(f"  Host IP: {pod.status.host_ip}")
print(f"  Start time: {pod.status.start_time}")
```

### Label Selectors and Field Selectors

Label selectors are how Kubernetes (and your automation) finds specific resources:

```python
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

# Find pods with a specific label
pods = v1.list_namespaced_pod(
    namespace="default",
    label_selector="app=web-server"
)
print(f"Pods with app=web-server: {len(pods.items)}")

# Multiple label selectors
pods = v1.list_namespaced_pod(
    namespace="default",
    label_selector="app=web-server,env=production"
)

# Label selector with set-based requirement
pods = v1.list_namespaced_pod(
    namespace="default",
    label_selector="app in (web-server, api-server),env!=test"
)

# Field selectors (filter by status, node, etc.)
running_pods = v1.list_namespaced_pod(
    namespace="default",
    field_selector="status.phase=Running"
)
print(f"Running pods: {len(running_pods.items)}")

# Pods on a specific node
pods_on_node = v1.list_pod_for_all_namespaces(
    field_selector="spec.nodeName=worker-node-1"
)
print(f"Pods on worker-node-1: {len(pods_on_node.items)}")
```

### Executing Commands in Pods

```python
from kubernetes import client, config
from kubernetes.stream import stream

config.load_kube_config()
v1 = client.CoreV1Api()

# Execute a command in a pod (like kubectl exec)
result = stream(
    v1.connect_get_namespaced_pod_exec,
    name="my-pod",
    namespace="default",
    command=["sh", "-c", "hostname && date && cat /etc/os-release"],
    stderr=True,
    stdin=False,
    stdout=True,
    tty=False,
)
print(result)

# Execute in a specific container
result = stream(
    v1.connect_get_namespaced_pod_exec,
    name="my-pod",
    namespace="default",
    container="app",
    command=["python", "-c", "import sys; print(sys.version)"],
    stderr=True,
    stdin=False,
    stdout=True,
    tty=False,
)
print(result)
```

## Working with Deployments

Deployments manage replica sets and rolling updates. The Python client lets you create, update, and scale them.

### Listing and Inspecting Deployments

```python
from kubernetes import client, config

config.load_kube_config()
apps_v1 = client.AppsV1Api()

# List deployments
deployments = apps_v1.list_namespaced_deployment(namespace="default")

for dep in deployments.items:
    name = dep.metadata.name
    replicas = dep.spec.replicas
    ready = dep.status.ready_replicas or 0
    available = dep.status.available_replicas or 0
    updated = dep.status.updated_replicas or 0

    print(f"  {name:35s}  {ready}/{replicas} ready  "
          f"{available} available  {updated} updated")

# Get a specific deployment
dep = apps_v1.read_namespaced_deployment(name="web-app", namespace="default")
print(f"\nDeployment: {dep.metadata.name}")
print(f"  Replicas: {dep.spec.replicas}")
print(f"  Strategy: {dep.spec.strategy.type}")
print(f"  Image: {dep.spec.template.spec.containers[0].image}")
```

### Creating a Deployment

```python
from kubernetes import client, config

config.load_kube_config()
apps_v1 = client.AppsV1Api()

# Create a deployment programmatically
deployment = client.V1Deployment(
    metadata=client.V1ObjectMeta(
        name="web-app",
        labels={"app": "web-app"},
    ),
    spec=client.V1DeploymentSpec(
        replicas=3,
        selector=client.V1LabelSelector(
            match_labels={"app": "web-app"},
        ),
        template=client.V1PodTemplateSpec(
            metadata=client.V1ObjectMeta(
                labels={"app": "web-app"},
            ),
            spec=client.V1PodSpec(
                containers=[
                    client.V1Container(
                        name="web",
                        image="nginx:1.25",
                        ports=[client.V1ContainerPort(container_port=80)],
                        resources=client.V1ResourceRequirements(
                            requests={"cpu": "100m", "memory": "128Mi"},
                            limits={"cpu": "500m", "memory": "256Mi"},
                        ),
                        liveness_probe=client.V1Probe(
                            http_get=client.V1HTTPGetAction(path="/", port=80),
                            initial_delay_seconds=10,
                            period_seconds=5,
                        ),
                    ),
                ],
            ),
        ),
    ),
)

result = apps_v1.create_namespaced_deployment(
    namespace="default",
    body=deployment,
)
print(f"Created deployment: {result.metadata.name}")
```

### Scaling a Deployment

```python
from kubernetes import client, config

config.load_kube_config()
apps_v1 = client.AppsV1Api()


def scale_deployment(name, namespace, replicas):
    """Scale a deployment to the specified number of replicas."""
    # Patch the deployment's replica count
    body = {"spec": {"replicas": replicas}}
    result = apps_v1.patch_namespaced_deployment_scale(
        name=name,
        namespace=namespace,
        body=body,
    )
    print(f"Scaled {name} to {result.spec.replicas} replicas")
    return result


# Scale up
scale_deployment("web-app", "default", 5)

# Scale down
scale_deployment("web-app", "default", 2)

# Scale to zero (useful for maintenance)
scale_deployment("web-app", "default", 0)
```

### Updating a Deployment (Rolling Update)

```python
from kubernetes import client, config

config.load_kube_config()
apps_v1 = client.AppsV1Api()


def update_deployment_image(name, namespace, container_name, new_image):
    """Update a deployment's container image (triggers rolling update)."""
    # Read current deployment
    deployment = apps_v1.read_namespaced_deployment(name=name, namespace=namespace)

    # Find the container and update its image
    for container in deployment.spec.template.spec.containers:
        if container.name == container_name:
            old_image = container.image
            container.image = new_image
            break
    else:
        raise ValueError(f"Container '{container_name}' not found in deployment '{name}'")

    # Apply the update
    result = apps_v1.replace_namespaced_deployment(
        name=name,
        namespace=namespace,
        body=deployment,
    )

    print(f"Updated {name}/{container_name}: {old_image} -> {new_image}")
    return result


# Trigger a rolling update
update_deployment_image("web-app", "default", "web", "nginx:1.26")
```

## Working with Services, Namespaces, and Nodes

```python
from kubernetes import client, config

config.load_kube_config()
v1 = client.CoreV1Api()

# --- Services ---
services = v1.list_namespaced_service(namespace="default")
for svc in services.items:
    svc_type = svc.spec.type
    cluster_ip = svc.spec.cluster_ip
    ports = ", ".join(
        f"{p.port}/{p.protocol}" for p in (svc.spec.ports or [])
    )
    print(f"  {svc.metadata.name:30s}  {svc_type:12s}  {cluster_ip:16s}  {ports}")

# --- Namespaces ---
namespaces = v1.list_namespace()
for ns in namespaces.items:
    print(f"  {ns.metadata.name:30s}  {ns.status.phase}")

# --- Nodes ---
nodes = v1.list_node()
for node in nodes.items:
    # Get node conditions
    conditions = {c.type: c.status for c in node.status.conditions}
    ready = conditions.get("Ready", "Unknown")

    # Get resource capacity
    capacity = node.status.capacity
    cpu = capacity.get("cpu", "?")
    memory = capacity.get("memory", "?")

    # Get node labels for role
    labels = node.metadata.labels or {}
    roles = [k.split("/")[-1] for k in labels if "node-role.kubernetes.io/" in k]
    role = ", ".join(roles) if roles else "worker"

    print(f"  {node.metadata.name:30s}  Ready={ready:5s}  "
          f"CPU={cpu:4s}  Memory={memory:12s}  Role={role}")
```

## Creating Resources from YAML

Sometimes it is easier to define Kubernetes resources in YAML (especially for complex objects) and create them with Python:

```python
from kubernetes import client, config, utils
import yaml

config.load_kube_config()
k8s_client = client.ApiClient()

# Load a YAML manifest
yaml_content = """
apiVersion: v1
kind: ConfigMap
metadata:
  name: app-config
  namespace: default
data:
  DATABASE_URL: "postgresql://db:5432/appdb"
  REDIS_URL: "redis://cache:6379"
  LOG_LEVEL: "info"
---
apiVersion: v1
kind: Service
metadata:
  name: web-service
  namespace: default
spec:
  selector:
    app: web-app
  ports:
    - port: 80
      targetPort: 8080
  type: ClusterIP
"""

# Create all resources from the YAML
resources = yaml.safe_load_all(yaml_content)
for resource in resources:
    try:
        utils.create_from_dict(k8s_client, resource)
        kind = resource.get("kind", "unknown")
        name = resource.get("metadata", {}).get("name", "unknown")
        print(f"Created {kind}: {name}")
    except utils.FailToCreateError as e:
        print(f"Failed: {e}")

# Create from a YAML file
# utils.create_from_yaml(k8s_client, "deployment.yaml")

# Create from a directory of YAML files
# utils.create_from_yaml(k8s_client, "k8s-manifests/", verbose=True)
```

<Callout type="warning" title="Idempotency Matters">
`create_from_dict` will fail if the resource already exists. For automation that might run multiple times, use the patch or replace API methods instead, or catch the `409 Conflict` error and handle it. Production automation should always be safe to run twice.
</Callout>

## Watching Events

The Kubernetes watch API lets you monitor the cluster in real-time. Instead of polling, you get notified when resources change.

```python
from kubernetes import client, config, watch

config.load_kube_config()
v1 = client.CoreV1Api()

# Watch for pod events in a namespace
w = watch.Watch()

print("Watching pods (Ctrl+C to stop)...")
try:
    for event in w.stream(v1.list_namespaced_pod, namespace="default", timeout_seconds=60):
        event_type = event["type"]        # ADDED, MODIFIED, DELETED
        pod = event["object"]
        name = pod.metadata.name
        phase = pod.status.phase

        print(f"  [{event_type:8s}] {name:40s} -> {phase}")

        # Stop watching after a specific condition
        if event_type == "DELETED" and "cleanup" in name:
            w.stop()
            break
except KeyboardInterrupt:
    w.stop()

# Watch for events across all namespaces
print("\nWatching cluster events...")
w = watch.Watch()
for event in w.stream(v1.list_event_for_all_namespaces, timeout_seconds=30):
    ev = event["object"]
    kind = ev.involved_object.kind
    name = ev.involved_object.name
    reason = ev.reason
    message = ev.message
    print(f"  {kind:15s} {name:30s} {reason:20s} {message}")
```

## Error Handling

The Kubernetes API raises specific exceptions that your automation must handle:

```python
from kubernetes import client, config
from kubernetes.client.exceptions import ApiException
import json

config.load_kube_config()
v1 = client.CoreV1Api()
apps_v1 = client.AppsV1Api()


def safe_get_pod(name, namespace="default"):
    """Get a pod, returning None if not found."""
    try:
        return v1.read_namespaced_pod(name=name, namespace=namespace)
    except ApiException as e:
        if e.status == 404:
            print(f"Pod not found: {name}")
            return None
        elif e.status == 403:
            print(f"Permission denied: cannot read pod {name}")
            return None
        else:
            print(f"API error ({e.status}): {e.reason}")
            raise


def safe_create_deployment(deployment, namespace="default"):
    """Create a deployment, handling conflicts."""
    try:
        result = apps_v1.create_namespaced_deployment(
            namespace=namespace,
            body=deployment,
        )
        print(f"Created deployment: {result.metadata.name}")
        return result
    except ApiException as e:
        if e.status == 409:
            # Already exists — update instead
            name = deployment.metadata.name
            print(f"Deployment {name} already exists, updating...")
            result = apps_v1.replace_namespaced_deployment(
                name=name,
                namespace=namespace,
                body=deployment,
            )
            print(f"Updated deployment: {result.metadata.name}")
            return result
        elif e.status == 422:
            # Unprocessable entity — usually validation error
            body = json.loads(e.body)
            print(f"Validation error: {body.get('message', e.reason)}")
            return None
        else:
            print(f"API error ({e.status}): {e.reason}")
            raise


def safe_delete_resource(api_func, name, namespace="default"):
    """Delete a resource, ignoring 'not found' errors."""
    try:
        api_func(name=name, namespace=namespace)
        print(f"Deleted: {name}")
        return True
    except ApiException as e:
        if e.status == 404:
            print(f"Already gone: {name}")
            return True
        else:
            print(f"Failed to delete {name}: {e.reason}")
            return False


# Usage
safe_delete_resource(v1.delete_namespaced_pod, "old-pod")
safe_delete_resource(apps_v1.delete_namespaced_deployment, "old-deployment")
```

<Exercise number={1} title="Pod Inspector Script">

Write a script that lists all pods across namespaces, groups them by status, and reports any pods in a problematic state (CrashLoopBackOff, ImagePullBackOff, Pending for too long).

```python
#!/usr/bin/env python3
"""Pod inspector — finds and reports problematic pods."""

from kubernetes import client, config
from kubernetes.client.exceptions import ApiException
from datetime import datetime, timezone, timedelta
from collections import defaultdict

# Load config
try:
    config.load_incluster_config()
except config.ConfigException:
    config.load_kube_config()

v1 = client.CoreV1Api()


def get_pod_condition(pod):
    """Determine the effective condition of a pod."""
    phase = pod.status.phase or "Unknown"
    container_statuses = pod.status.container_statuses or []

    # Check container-level issues
    for cs in container_statuses:
        if cs.state.waiting:
            reason = cs.state.waiting.reason or "Waiting"
            if reason in ("CrashLoopBackOff", "ImagePullBackOff",
                          "ErrImagePull", "CreateContainerConfigError"):
                return reason
        if cs.state.terminated:
            reason = cs.state.terminated.reason or "Terminated"
            if reason not in ("Completed",):
                return f"Terminated:{reason}"

    return phase


def get_pod_age(pod):
    """Get pod age as a human-readable string."""
    if not pod.metadata.creation_timestamp:
        return "unknown"

    now = datetime.now(timezone.utc)
    age = now - pod.metadata.creation_timestamp.replace(tzinfo=timezone.utc)

    if age.days > 0:
        return f"{age.days}d"
    hours = age.seconds // 3600
    if hours > 0:
        return f"{hours}h"
    minutes = age.seconds // 60
    return f"{minutes}m"


def inspect_pods():
    """Inspect all pods and generate a report."""
    print("=" * 80)
    print("  Pod Inspector Report")
    print(f"  Generated: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
    print("=" * 80)

    try:
        pods = v1.list_pod_for_all_namespaces()
    except ApiException as e:
        print(f"Error listing pods: {e.reason}")
        return

    # Group pods by condition
    by_condition = defaultdict(list)
    by_namespace = defaultdict(int)
    problems = []

    for pod in pods.items:
        condition = get_pod_condition(pod)
        by_condition[condition].append(pod)
        by_namespace[pod.metadata.namespace] += 1

        # Flag problematic pods
        problem_conditions = {
            "CrashLoopBackOff", "ImagePullBackOff", "ErrImagePull",
            "CreateContainerConfigError", "Failed", "Unknown",
        }
        if condition in problem_conditions or condition.startswith("Terminated:"):
            problems.append(pod)

        # Flag pods pending for too long
        if condition == "Pending" and pod.metadata.creation_timestamp:
            age = datetime.now(timezone.utc) - pod.metadata.creation_timestamp.replace(
                tzinfo=timezone.utc
            )
            if age > timedelta(minutes=5):
                problems.append(pod)

    # Summary by condition
    print(f"\n--- Pod Status Summary ---")
    total = len(pods.items)
    print(f"Total pods: {total}\n")
    for condition, pod_list in sorted(by_condition.items(), key=lambda x: -len(x[1])):
        pct = len(pod_list) / total * 100 if total else 0
        bar = "#" * int(pct / 2)
        print(f"  {condition:30s}  {len(pod_list):4d}  ({pct:5.1f}%)  {bar}")

    # Summary by namespace
    print(f"\n--- Pods by Namespace ---")
    for ns, count in sorted(by_namespace.items(), key=lambda x: -x[1]):
        print(f"  {ns:30s}  {count:4d}")

    # Problem pods
    if problems:
        print(f"\n--- Problematic Pods ({len(problems)}) ---")
        print(f"{'NAMESPACE':20s} {'POD':35s} {'CONDITION':25s} {'AGE':6s} {'RESTARTS'}")
        print("-" * 95)

        for pod in problems:
            ns = pod.metadata.namespace
            name = pod.metadata.name
            condition = get_pod_condition(pod)
            age = get_pod_age(pod)
            restarts = sum(
                cs.restart_count for cs in (pod.status.container_statuses or [])
            )
            print(f"{ns:20s} {name:35s} {condition:25s} {age:6s} {restarts}")

        # Detailed info for each problem pod
        print(f"\n--- Problem Details ---")
        for pod in problems[:5]:  # Limit detail to first 5
            name = f"{pod.metadata.namespace}/{pod.metadata.name}"
            print(f"\n  {name}")
            for cs in (pod.status.container_statuses or []):
                if cs.state.waiting:
                    print(f"    Container '{cs.name}': Waiting — {cs.state.waiting.reason}")
                    if cs.state.waiting.message:
                        print(f"      Message: {cs.state.waiting.message}")
                if cs.state.terminated:
                    print(f"    Container '{cs.name}': Terminated — {cs.state.terminated.reason}")
                    print(f"      Exit code: {cs.state.terminated.exit_code}")
    else:
        print(f"\n--- No Problematic Pods Found ---")
        print("  All pods are healthy.")

    print()


if __name__ == "__main__":
    inspect_pods()
```

<Exercise number={2} title="Deployment Manager">

Write a script that manages deployments: list them with status, scale by name, and update the container image to trigger a rolling update.

```python
#!/usr/bin/env python3
"""Deployment manager — list, scale, and update deployments."""

from kubernetes import client, config
from kubernetes.client.exceptions import ApiException
import sys
import time

# Load config
try:
    config.load_incluster_config()
except config.ConfigException:
    config.load_kube_config()

apps_v1 = client.AppsV1Api()
v1 = client.CoreV1Api()


def list_deployments(namespace="default"):
    """List all deployments with their current status."""
    try:
        deployments = apps_v1.list_namespaced_deployment(namespace=namespace)
    except ApiException as e:
        print(f"Error: {e.reason}")
        return

    if not deployments.items:
        print(f"No deployments found in namespace '{namespace}'")
        return

    print(f"\nDeployments in '{namespace}':")
    print(f"{'NAME':35s} {'READY':10s} {'UP-TO-DATE':12s} {'AVAILABLE':12s} {'IMAGE'}")
    print("-" * 110)

    for dep in deployments.items:
        name = dep.metadata.name
        replicas = dep.spec.replicas or 0
        ready = dep.status.ready_replicas or 0
        updated = dep.status.updated_replicas or 0
        available = dep.status.available_replicas or 0
        image = dep.spec.template.spec.containers[0].image

        ready_str = f"{ready}/{replicas}"
        print(f"{name:35s} {ready_str:10s} {updated:<12d} {available:<12d} {image}")

    print()


def scale_deployment(name, replicas, namespace="default"):
    """Scale a deployment to the specified number of replicas."""
    print(f"Scaling {name} to {replicas} replicas...")

    try:
        # Read current state
        dep = apps_v1.read_namespaced_deployment(name=name, namespace=namespace)
        old_replicas = dep.spec.replicas

        # Patch the scale
        body = {"spec": {"replicas": replicas}}
        result = apps_v1.patch_namespaced_deployment_scale(
            name=name,
            namespace=namespace,
            body=body,
        )

        print(f"  {name}: {old_replicas} -> {result.spec.replicas} replicas")

        # Wait for the scaling to complete
        print("  Waiting for rollout...", end="", flush=True)
        for _ in range(30):
            dep = apps_v1.read_namespaced_deployment(name=name, namespace=namespace)
            ready = dep.status.ready_replicas or 0
            if ready == replicas:
                print(f" done ({ready}/{replicas} ready)")
                return True
            print(".", end="", flush=True)
            time.sleep(2)

        print(f" timeout (current: {dep.status.ready_replicas or 0}/{replicas})")
        return False

    except ApiException as e:
        if e.status == 404:
            print(f"  Deployment '{name}' not found in namespace '{namespace}'")
        else:
            print(f"  Error: {e.reason}")
        return False


def update_image(name, container_name, new_image, namespace="default"):
    """Update a deployment's container image, triggering a rolling update."""
    print(f"Updating {name}/{container_name} to {new_image}...")

    try:
        dep = apps_v1.read_namespaced_deployment(name=name, namespace=namespace)

        # Find the container
        container_found = False
        for container in dep.spec.template.spec.containers:
            if container.name == container_name:
                old_image = container.image
                container.image = new_image
                container_found = True
                break

        if not container_found:
            print(f"  Container '{container_name}' not found in deployment '{name}'")
            containers = [c.name for c in dep.spec.template.spec.containers]
            print(f"  Available containers: {containers}")
            return False

        # Apply the update
        apps_v1.replace_namespaced_deployment(
            name=name, namespace=namespace, body=dep,
        )
        print(f"  Image updated: {old_image} -> {new_image}")

        # Monitor the rollout
        print("  Monitoring rollout...", end="", flush=True)
        desired = dep.spec.replicas or 1
        for _ in range(60):
            dep = apps_v1.read_namespaced_deployment(name=name, namespace=namespace)
            updated = dep.status.updated_replicas or 0
            ready = dep.status.ready_replicas or 0
            available = dep.status.available_replicas or 0

            if updated == desired and ready == desired and available == desired:
                print(f" complete ({ready}/{desired} ready)")
                return True

            print(".", end="", flush=True)
            time.sleep(2)

        print(f" timeout — check 'kubectl rollout status deployment/{name}'")
        return False

    except ApiException as e:
        if e.status == 404:
            print(f"  Deployment '{name}' not found")
        elif e.status == 422:
            print(f"  Invalid update: {e.reason}")
        else:
            print(f"  Error: {e.reason}")
        return False


def show_deployment_detail(name, namespace="default"):
    """Show detailed information about a deployment."""
    try:
```

```python
def format_age(timestamp):
    """Convert a timestamp to a human-readable age string."""
    if not timestamp:
        return "N/A"
    now = datetime.now(timezone.utc)
    ts = timestamp.replace(tzinfo=timezone.utc) if timestamp.tzinfo is None else timestamp
    delta = now - ts
    if delta.days > 0:
        return f"{delta.days}d"
    hours = delta.seconds // 3600
    if hours > 0:
        return f"{hours}h"
    minutes = delta.seconds // 60
    return f"{minutes}m"


def format_bytes(value_str):
    """Convert Kubernetes memory strings (Ki, Mi, Gi) to readable format."""
    if not value_str:
        return "N/A"
    if value_str.endswith("Ki"):
        return f"{int(value_str[:-2]) // 1024} Mi"
    if value_str.endswith("Mi"):
        return f"{value_str}"
    if value_str.endswith("Gi"):
        return f"{value_str}"
    return value_str
```

## Key Takeaways

- Use `config.load_kube_config()` for local development and `config.load_incluster_config()` when running inside a cluster
- The Kubernetes API is organized by groups: `CoreV1Api` for pods/services, `AppsV1Api` for deployments, `BatchV1Api` for jobs
- Label selectors are the primary way to find and filter resources — always label your workloads
- The `watch` API is far more efficient than polling — use it for real-time monitoring
- Always handle `ApiException` with status code checks: 404 (not found), 409 (conflict), 403 (forbidden), 422 (validation)
- Creating resources from YAML with `create_from_dict` is simpler than building Python objects for complex manifests
- Scale deployments with `patch_namespaced_deployment_scale` — it is cleaner than patching the full deployment
- Pod logs and exec require the pod to be running — check status before attempting these operations
- For idempotent automation, try create and fall back to replace on 409 Conflict
