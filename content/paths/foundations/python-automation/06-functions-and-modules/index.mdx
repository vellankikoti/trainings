### Return Values

```python
def check_server(hostname, port=80):
    """Check server health and return a result dict."""
    import socket
    import time

    start = time.time()
    try:
        with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:
            sock.settimeout(5)
            result = sock.connect_ex((hostname, port))
            elapsed = (time.time() - start) * 1000

            return {
                "hostname": hostname,
                "port": port,
                "is_up": result == 0,
                "response_ms": round(elapsed, 2),
                "error": None,
            }
    except socket.error as e:
        return {
            "hostname": hostname,
            "port": port,
            "is_up": False,
            "response_ms": None,
            "error": str(e),
        }

# Use the result
result = check_server("localhost", 22)
if result["is_up"]:
    print(f"{result['hostname']} is up ({result['response_ms']}ms)")
else:
    print(f"{result['hostname']} is down: {result['error']}")
```

## Decorators

Decorators wrap functions to add behavior without modifying the original function. They're perfect for cross-cutting concerns like retry logic, timing, and logging.

### Timer Decorator

```python
import time
import functools

def timer(func):
    """Measure and print execution time of a function."""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        elapsed = time.time() - start
        print(f"  [{func.__name__}] completed in {elapsed:.3f}s")
        return result
    return wrapper

@timer
def deploy_service(service):
    """Deploy a service (simulated)."""
    time.sleep(0.5)  # Simulate work
    return {"status": "deployed"}

# When you call deploy_service(), the timer decorator runs automatically
result = deploy_service("web-api")
# [deploy_service] completed in 0.502s
```

### Retry Decorator

```python
import time
import functools

def retry(max_attempts: int = 3, delay: float = 1.0, backoff: float = 2.0):
    """Retry a function on failure with exponential backoff.

    Args:
        max_attempts: Maximum number of attempts.
        delay: Initial delay between retries in seconds.
        backoff: Multiplier for delay after each retry.
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            current_delay = delay
            last_exception = None

            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    if attempt == max_attempts:
                        print(f"  [{func.__name__}] Failed after {max_attempts} attempts: {e}")
                        raise
                    print(f"  [{func.__name__}] Attempt {attempt} failed: {e}. "
                          f"Retrying in {current_delay}s...")
                    time.sleep(current_delay)
                    current_delay *= backoff

        return wrapper
    return decorator

@retry(max_attempts=3, delay=1.0, backoff=2.0)
def call_api(url):
    """Call an API endpoint (may fail)."""
    import requests
    response = requests.get(url, timeout=5)
    response.raise_for_status()
    return response.json()

# Usage — retries automatically on failure
# result = call_api("https://api.example.com/health")
```

### Logging Decorator

```python
import functools
import logging

logger = logging.getLogger(__name__)

def log_call(func):
    """Log function calls with arguments and return values."""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        args_repr = [repr(a) for a in args]
        kwargs_repr = [f"{k}={v!r}" for k, v in kwargs.items()]
        signature = ", ".join(args_repr + kwargs_repr)
        logger.info(f"Calling {func.__name__}({signature})")

        try:
            result = func(*args, **kwargs)
            logger.info(f"{func.__name__} returned {result!r}")
            return result
        except Exception as e:
            logger.error(f"{func.__name__} raised {type(e).__name__}: {e}")
            raise

    return wrapper

@log_call
def restart_service(service_name, force=False):
    """Restart a service."""
    return {"service": service_name, "status": "restarted"}
```

<Callout type="warning" title="Always Use @functools.wraps">
Without `@functools.wraps(func)`, the decorator replaces the function's `__name__` and `__doc__` attributes. This breaks debugging, logging, and documentation. Always include it.
</Callout>

Create a configurable retry decorator and test it with a flaky function:

```python
#!/usr/bin/env python3
"""Build and test a retry decorator."""

import time
import functools
import random

def retry(max_attempts=3, delay=1.0, backoff=2.0, exceptions=(Exception,)):
    """Retry decorator with exponential backoff.

    Args:
        max_attempts: Maximum retry attempts.
        delay: Initial delay between retries (seconds).
        backoff: Multiplier applied to delay after each retry.
        exceptions: Tuple of exception types to catch.
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            current_delay = delay
            for attempt in range(1, max_attempts + 1):
                try:
                    result = func(*args, **kwargs)
                    if attempt > 1:
                        print(f"  [{func.__name__}] Succeeded on attempt {attempt}")
                    return result
                except exceptions as e:
                    if attempt == max_attempts:
                        print(f"  [{func.__name__}] Failed after {max_attempts} attempts")
                        raise
                    print(f"  [{func.__name__}] Attempt {attempt} failed: {e}. "
                          f"Retrying in {current_delay:.1f}s...")
                    time.sleep(current_delay)
                    current_delay *= backoff
        return wrapper
    return decorator

# Test with a flaky function that fails randomly
call_count = 0

@retry(max_attempts=5, delay=0.1, backoff=1.5)
def flaky_api_call(endpoint):
    """Simulate an unreliable API call."""
    global call_count
    call_count += 1
    if random.random() < 0.6:  # 60% chance of failure
        raise ConnectionError(f"Connection to {endpoint} refused")
    return {"status": "ok", "endpoint": endpoint}

# Test the retry decorator
print("Testing retry decorator:\n")
for i in range(3):
    call_count = 0
    try:
        result = flaky_api_call(f"/api/test/{i}")
        print(f"  Result: {result}\n")
    except ConnectionError:
        print(f"  Gave up after all retries\n")
```

```python
#!/usr/bin/env python3
"""Standard library utilities for DevOps."""

import os
import sys
import subprocess
import json
from pathlib import Path
from datetime import datetime, timedelta
from collections import Counter, defaultdict

# 1. Run a command and parse its output
def run_command(cmd, timeout=30):
    """Run a shell command and return (success, stdout, stderr)."""
    try:
        result = subprocess.run(
            cmd, shell=True, capture_output=True, text=True, timeout=timeout
        )
        return result.returncode == 0, result.stdout.strip(), result.stderr.strip()
    except subprocess.TimeoutExpired:
        return False, "", f"Command timed out after {timeout}s"

# 2. Find files modified in the last N hours
def find_recent_files(directory, hours=24, pattern="*"):
    """Find files modified within the last N hours."""
    cutoff = datetime.now() - timedelta(hours=hours)
    path = Path(directory)
    recent = []

    if not path.exists():
        return recent

    for file_path in path.rglob(pattern):
        if file_path.is_file():
            mod_time = datetime.fromtimestamp(file_path.stat().st_mtime)
            if mod_time > cutoff:
                recent.append({
                    "path": str(file_path),
                    "size": file_path.stat().st_size,
                    "modified": mod_time.isoformat(),
                })

    return sorted(recent, key=lambda f: f["modified"], reverse=True)

# 3. Parse environment into a config dict
def get_app_config(prefix="APP_"):
    """Extract all environment variables with a given prefix into a config dict."""
    config = {}
    for key, value in os.environ.items():
        if key.startswith(prefix):
            clean_key = key[len(prefix):].lower()
            # Try to convert types
            if value.lower() in ("true", "false"):
                config[clean_key] = value.lower() == "true"
            elif value.isdigit():
                config[clean_key] = int(value)
            else:
                config[clean_key] = value
    return config

# 4. Summarize disk usage by directory
def disk_usage_summary(directory, depth=1):
    """Summarize disk usage by subdirectory."""
    path = Path(directory)
    if not path.exists():
        return {}

    usage = defaultdict(int)
    for file_path in path.rglob("*"):
        if file_path.is_file():
            # Get the relative path at the desired depth
            try:
                relative = file_path.relative_to(path)
                parts = relative.parts
                key = str(Path(*parts[:depth])) if len(parts) > depth else str(relative)
                usage[key] += file_path.stat().st_size
            except (ValueError, PermissionError):
                continue

    return dict(sorted(usage.items(), key=lambda x: x[1], reverse=True))

# Demo
print("=== DevOps Standard Library Utilities ===\n")

# Test run_command
print("[1] Running system commands:")
success, stdout, stderr = run_command("uname -a")
print(f"  uname: {stdout[:80]}...")

success, stdout, stderr = run_command("python3 --version")
print(f"  Python: {stdout}")

# Test find_recent_files
print(f"\n[2] Recent Python files in current directory:")
recent = find_recent_files(".", hours=48, pattern="*.py")
for f in recent[:5]:
    print(f"  {f['path']} ({f['size']} bytes) modified {f['modified']}")

# Test config parser
print(f"\n[3] App configuration from environment:")
os.environ.update({"APP_PORT": "8080", "APP_DEBUG": "true", "APP_NAME": "demo"})
config = get_app_config()
print(f"  Config: {json.dumps(config, indent=4)}")

# Test disk usage
print(f"\n[4] Disk usage summary:")
usage = disk_usage_summary(".", depth=1)
for path_name, size in list(usage.items())[:5]:
    size_kb = size / 1024
    print(f"  {path_name:<30} {size_kb:>10.1f} KB")
```

<MiniProject title="DevOps Utility Library">

Build a reusable DevOps utility library with decorators, helper functions, and a clean module structure:

```python
#!/usr/bin/env python3
"""devops_toolkit.py — A reusable DevOps utility library.

Provides decorators (retry, timer, log_call), notification helpers,
and common DevOps utility functions. Designed to be imported by
automation scripts.

Usage:
    from devops_toolkit import retry, timer, notify
    # or run directly for self-test:
    python3 devops_toolkit.py
"""

import functools
import logging
import os
import socket
import subprocess
import sys
import time
from datetime import datetime
from pathlib import Path
from typing import Optional

# ============================================================
# Logging Setup
# ============================================================

def setup_logging(
    level: str = "INFO",
    log_format: str = "%(asctime)s [%(levelname)s] %(name)s: %(message)s",
    log_file: Optional[str] = None,
) -> logging.Logger:
    """Configure logging for a DevOps script.

    Args:
        level: Log level (DEBUG, INFO, WARNING, ERROR).
        log_format: Format string for log messages.
        log_file: Optional file path to write logs to.

    Returns:
        Configured root logger.
    """
    logger = logging.getLogger()
    logger.setLevel(getattr(logging, level.upper(), logging.INFO))

    # Console handler
    console = logging.StreamHandler()
    console.setFormatter(logging.Formatter(log_format, datefmt="%Y-%m-%d %H:%M:%S"))
    logger.addHandler(console)

    # File handler (optional)
    if log_file:
        file_handler = logging.FileHandler(log_file)
        file_handler.setFormatter(logging.Formatter(log_format, datefmt="%Y-%m-%d %H:%M:%S"))
        logger.addHandler(file_handler)

    return logger


# ============================================================
# Decorators
# ============================================================

def timer(func):
    """Measure and log execution time."""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        logger = logging.getLogger(func.__module__)
        start = time.time()
        try:
            result = func(*args, **kwargs)
            elapsed = time.time() - start
            logger.info(f"{func.__name__} completed in {elapsed:.3f}s")
            return result
        except Exception:
            elapsed = time.time() - start
            logger.error(f"{func.__name__} failed after {elapsed:.3f}s")
            raise
    return wrapper


def retry(
    max_attempts: int = 3,
    delay: float = 1.0,
    backoff: float = 2.0,
    exceptions: tuple = (Exception,),
):
    """Retry a function with exponential backoff.

    Args:
        max_attempts: Maximum number of attempts.
        delay: Initial delay between retries (seconds).
        backoff: Multiplier for delay after each retry.
        exceptions: Exception types to catch and retry on.
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            logger = logging.getLogger(func.__module__)
            current_delay = delay

            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    if attempt == max_attempts:
                        logger.error(
                            f"{func.__name__} failed after {max_attempts} attempts: {e}"
                        )
                        raise
                    logger.warning(
                        f"{func.__name__} attempt {attempt}/{max_attempts} failed: {e}. "
                        f"Retrying in {current_delay:.1f}s..."
                    )
                    time.sleep(current_delay)
                    current_delay *= backoff

        return wrapper
    return decorator


def log_call(func):
    """Log function calls with arguments and results."""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        logger = logging.getLogger(func.__module__)
        args_str = ", ".join(
            [repr(a) for a in args] +
            [f"{k}={v!r}" for k, v in kwargs.items()]
        )
        logger.debug(f"Calling {func.__name__}({args_str})")

        try:
            result = func(*args, **kwargs)
            logger.debug(f"{func.__name__} returned {result!r}")
            return result
        except Exception as e:
            logger.error(f"{func.__name__} raised {type(e).__name__}: {e}")
            raise

    return wrapper

def check_port(host: str, port: int, timeout: float = 3.0) -> bool:
    """Check if a TCP port is accepting connections."""
    try:
        with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:
            sock.settimeout(timeout)
            return sock.connect_ex((host, port)) == 0
    except socket.error:
        return False


def run_command(
    command: str,
    timeout: int = 30,
    check: bool = False,
) -> dict:
    """Run a shell command and return structured results.

    Args:
        command: Shell command to execute.
        timeout: Maximum execution time in seconds.
        check: If True, raise on non-zero exit code.

    Returns:
        Dict with keys: success, exit_code, stdout, stderr, duration_ms.
    """
    start = time.time()
    try:
        result = subprocess.run(
            command, shell=True, capture_output=True,
            text=True, timeout=timeout,
        )
        duration_ms = (time.time() - start) * 1000

        output = {
            "success": result.returncode == 0,
            "exit_code": result.returncode,
            "stdout": result.stdout.strip(),
            "stderr": result.stderr.strip(),
            "duration_ms": round(duration_ms, 2),
        }

        if check and result.returncode != 0:
            raise subprocess.CalledProcessError(
                result.returncode, command, result.stdout, result.stderr
            )

        return output

    except subprocess.TimeoutExpired:
        return {
            "success": False,
            "exit_code": -1,
            "stdout": "",
            "stderr": f"Command timed out after {timeout}s",
            "duration_ms": timeout * 1000,
        }


def notify(message: str, level: str = "info", channel: str = "console"):
    """Send a notification. Supports console output and Slack webhook.

    Args:
        message: Notification message.
        level: Severity level (info, warning, error, critical).
        channel: Notification channel (console, slack).
    """
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    formatted = f"[{timestamp}] [{level.upper()}] {message}"

    if channel == "console":
        print(formatted)
    elif channel == "slack":
        webhook_url = os.environ.get("SLACK_WEBHOOK_URL")
        if not webhook_url:
            print(f"SLACK_WEBHOOK_URL not set — falling back to console")
            print(formatted)
            return

        import requests
        color_map = {"info": "#36a64f", "warning": "#ff9900",
                     "error": "#ff0000", "critical": "#8b0000"}
        payload = {
            "attachments": [{
                "color": color_map.get(level, "#808080"),
                "text": message,
                "footer": f"DevOps Toolkit | {timestamp}",
            }]
        }
        try:
            requests.post(webhook_url, json=payload, timeout=10)
        except Exception as e:
            print(f"Slack notification failed: {e}")
            print(formatted)


def wait_for_service(
    host: str,
    port: int,
    timeout: int = 60,
    interval: int = 5,
) -> bool:
    """Wait for a service to become available.

    Args:
        host: Hostname to check.
        port: Port to check.
        timeout: Maximum wait time in seconds.
        interval: Check interval in seconds.

    Returns:
        True if service became available, False if timed out.
    """
    logger = logging.getLogger(__name__)
    elapsed = 0

    while elapsed < timeout:
        if check_port(host, port):
            logger.info(f"{host}:{port} is ready (waited {elapsed}s)")
            return True
        logger.info(f"Waiting for {host}:{port}... ({elapsed}s/{timeout}s)")
        time.sleep(interval)
        elapsed += interval

    logger.error(f"{host}:{port} not ready after {timeout}s")
    return False
```

## Key Takeaways

- Functions should do one thing and have clear names — `check_port()`, not `do_stuff()`
- Type hints serve as documentation even without mypy — `def deploy(service: str) -> dict`
- Use Google-style docstrings for all public functions
- Decorators handle cross-cutting concerns: retry, timing, logging, authentication
- Always use `@functools.wraps(func)` in decorators to preserve function metadata
- `__name__ == "__main__"` lets a file be both a module and a standalone script
- Standard library essentials: `os`, `sys`, `pathlib`, `subprocess`, `datetime`, `collections`
